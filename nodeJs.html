Introduction à Node.js

Qu'est-ce que Node.js ?

Node.js est un environnement d'exécution JavaScript côté serveur basé sur le moteur JavaScript V8 de Chrome. Il permet de créer des applications réseau scalables et performantes.

Node.js utilise une architecture non-bloquante et orientée événements, ce qui le rend idéal pour les applications en temps réel et les API.

Installation et premier projet

Pour commencer avec Node.js :

$ node -v
$ npm -v
$ mkdir mon-projet-node
$ cd mon-projet-node
$ npm init -y

Création de serveurs

Serveur HTTP de base

const http = require('http');

const server = http.createServer((req, res) => {
  res.writeHead(200, {'Content-Type': 'text/plain'});
  res.end('Bonjour depuis Node.js!\n');
});

const PORT = 3000;
server.listen(PORT, () => {
  console.log(`Serveur en écoute sur http://localhost:${PORT}`);
});
État du serveur: arrêté

Express.js - Framework web

// Installation: npm install express
const express = require('express');
const app = express();
const PORT = 3000;

// Route simple
app.get('/', (req, res) => {
  res.send('Accueil Express!');
});

// Route avec paramètre
app.get('/utilisateur/:id', (req, res) => {
  res.send(`Utilisateur ID: ${req.params.id}`);
});

app.listen(PORT, () => {
  console.log(`Serveur Express sur port ${PORT}`);
});

Middleware Express

// Middleware pour journalisation
app.use((req, res, next) => {
  console.log(`${req.method} ${req.url}`);
  next(); // Passe au middleware suivant
});

// Middleware pour parser le JSON
app.use(express.json());

// Middleware pour fichiers statiques
app.use(express.static('public'));

CRUD avec Express et MongoDB

// Créer un utilisateur
app.post('/users', async (req, res) => {
  const user = new User(req.body);
  await user.save();
  res.status(201).send(user);
});

// Lire tous les utilisateurs
app.get('/users', async (req, res) => {
  const users = await User.find({});
  res.send(users);
});

// Mettre à jour un utilisateur
app.put('/users/:id', async (req, res) => {
  const user = await User.findByIdAndUpdate(req.params.id, req.body, { new: true });
  res.send(user);
});

// Supprimer un utilisateur
app.delete('/users/:id', async (req, res) => {
  await User.findByIdAndDelete(req.params.id);
  res.status(204).send();
});

Concepts avancés

Gestion des erreurs

// Middleware pour erreurs 404
app.use((req, res, next) => {
  res.status(404).send("Désolé, cette page n'existe pas!");
});

// Gestionnaire d'erreurs global
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Une erreur est survenue!');
});

// Avec async/await
app.get('/users/:id', async (req, res, next) => {
  try {
    const user = await User.findById(req.params.id);
    if (!user) {
      return res.status(404).send('Utilisateur non trouvé');
    }
    res.send(user);
  } catch (err) {
    next(err);
  }
});

Authentification avec JWT

// npm install jsonwebtoken bcrypt
const jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');

// Inscription
app.post('/register', async (req, res) => {
  const { email, password } = req.body;
  const hashedPassword = await bcrypt.hash(password, 10);
  const user = new User({ email, password: hashedPassword });
  await user.save();
  res.status(201).send('Utilisateur créé');
});

// Connexion
app.post('/login', async (req, res) => {
  const { email, password } = req.body;
  const user = await User.findOne({ email });
  if (!user) return res.status(400).send('Email ou mot de passe invalide');

  const validPassword = await bcrypt.compare(password, user.password);
  if (!validPassword) return res.status(400).send('Email ou mot de passe invalide');

  const token = jwt.sign({ _id: user._id }, 'SECRET_KEY');
  res.header('Authorization', token).send(token);
});

// Middleware d'authentification
function auth(req, res, next) {
  const token = req.header('Authorization');
  if (!token) return res.status(401).send('Accès refusé');

  try {
    const verified = jwt.verify(token, 'SECRET_KEY');
    req.user = verified;
    next();
  } catch (err) {
    res.status(400).send('Token invalide');
  }
}
