Introduction à MySQL

Qu'est-ce que MySQL ?

MySQL est un système de gestion de bases de données relationnelles (SGBDR) open-source. Il permet de stocker, organiser et récupérer des données de manière efficace et sécurisée.

MySQL est le SGBDR le plus populaire au monde, utilisé par des géants comme Facebook, Twitter, YouTube et Airbnb. Il fait partie de la stack LAMP (Linux, Apache, MySQL, PHP/Python/Perl).

Création de bases de données et tables

CREATE DATABASE ecole;
USE ecole;

CREATE TABLE etudiants (
  id INT PRIMARY KEY AUTO_INCREMENT,
  nom VARCHAR(50) NOT NULL,
  prenom VARCHAR(50) NOT NULL,
  email VARCHAR(100) UNIQUE,
  date_naissance DATE,
  moyenne DECIMAL(4,2) DEFAULT 0.00
);

Structure de la table "etudiants"

Champ Type Description
id INT Identifiant unique
nom VARCHAR(50) Nom de l'étudiant
prenom VARCHAR(50) Prénom de l'étudiant
email VARCHAR(100) Email unique
date_naissance DATE Date de naissance
moyenne DECIMAL(4,2) Moyenne générale

Requêtes SQL fondamentales

SELECT - Récupérer des données

-- Sélectionner toutes les colonnes
SELECT * FROM etudiants;

-- Sélectionner des colonnes spécifiques
SELECT nom, prenom, moyenne FROM etudiants;

-- Avec condition WHERE
SELECT * FROM etudiants WHERE moyenne > 15.0;

-- Trier avec ORDER BY
SELECT * FROM etudiants ORDER BY moyenne DESC;

UPDATE - Modifier des données

UPDATE etudiants
SET moyenne = 16.0
WHERE nom = 'Martin' AND prenom = 'Lucie';

DELETE - Supprimer des données

DELETE FROM etudiants
WHERE moyenne < 10.0;

Relations et Jointures

Clés primaires et étrangères

CREATE TABLE cours (
  id INT PRIMARY KEY AUTO_INCREMENT,
  titre VARCHAR(100) NOT NULL,
  professeur VARCHAR(50)
);

CREATE TABLE inscriptions (
  etudiant_id INT,
  cours_id INT,
  date_inscription DATE DEFAULT CURDATE(),
  PRIMARY KEY (etudiant_id, cours_id),
  FOREIGN KEY (etudiant_id) REFERENCES etudiants(id),
  FOREIGN KEY (cours_id) REFERENCES cours(id)
);

Jointures entre tables

-- INNER JOIN
SELECT e.nom, e.prenom, c.titre
FROM etudiants e
INNER JOIN inscriptions i ON e.id = i.etudiant_id
INNER JOIN cours c ON i.cours_id = c.id;

-- LEFT JOIN
SELECT e.nom, e.prenom, c.titre
FROM etudiants e
LEFT JOIN inscriptions i ON e.id = i.etudiant_id
LEFT JOIN cours c ON i.cours_id = c.id;

Requêtes avancées

Fonctions d'agrégation

-- Calculer la moyenne générale
SELECT AVG(moyenne) AS moyenne_generale FROM etudiants;

-- Trouver la note maximale
SELECT MAX(moyenne) AS meilleure_note FROM etudiants;

-- Compter le nombre d'étudiants
SELECT COUNT(*) AS nombre_etudiants FROM etudiants;

-- Regrouper par note
SELECT ROUND(moyenne) AS note_arrondie, COUNT(*)
FROM etudiants
GROUP BY note_arrondie
ORDER BY note_arrondie DESC;

Sous-requêtes

-- Étudiants avec une note supérieure à la moyenne
SELECT nom, prenom, moyenne
FROM etudiants
WHERE moyenne > (
  SELECT AVG(moyenne) FROM etudiants
);

-- Cours avec plus de 10 inscriptions
SELECT titre, professeur
FROM cours
WHERE id IN (
  SELECT cours_id
  FROM inscriptions
  GROUP BY cours_id
  HAVING COUNT(*) > 10
);

Administration MySQL

Gestion des utilisateurs

-- Créer un nouvel utilisateur
CREATE USER 'assistant'@'localhost' IDENTIFIED BY 'mdp_secret';

-- Accorder des privilèges
GRANT SELECT, INSERT, UPDATE ON ecole.* TO 'assistant'@'localhost';

-- Révoquer des privilèges
REVOKE DELETE ON ecole.* FROM 'assistant'@'localhost';

-- Voir les privilèges
SHOW GRANTS FOR 'assistant'@'localhost';
