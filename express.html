Introduction à Express.js

Qu'est-ce qu'Express.js ?

Express.js est un framework web minimaliste et flexible pour Node.js qui fournit un ensemble robuste de fonctionnalités pour développer des applications web et mobiles.

Express est le framework backend le plus populaire dans l'écosystème Node.js, servant de base à de nombreux autres frameworks.

Installation et configuration

$ npm init -y
$ npm install express
$ touch app.js

Premier serveur Express

// app.js
const express = require('express');
const app = express();
const PORT = 3000;

// Route de base
app.get('/', (req, res) => {
  res.send('Bienvenue sur mon serveur Express !');
});

// Démarrer le serveur
app.listen(PORT, () => {
  console.log(`Serveur Express démarré sur http://localhost:${PORT}`);
});

Routage et middleware

Gestion des routes

Express fournit des méthodes pour définir des routes correspondant aux verbes HTTP :

// Routes HTTP de base
app.get('/api/users', (req, res) => {
  res.json([{ id: 1, name: 'Alice' }, { id: 2, name: 'Bob' }]);
});

app.post('/api/users', (req, res) => {
  // Créer un nouvel utilisateur
  res.status(201).json({ message: 'Utilisateur créé' });
});

app.put('/api/users/:id', (req, res) => {
  const userId = req.params.id;
  res.json({ message: `Utilisateur ${userId} mis à jour` });
});

app.delete('/api/users/:id', (req, res) => {
  const userId = req.params.id;
  res.json({ message: `Utilisateur ${userId} supprimé` });
});

Middleware Express

Les middleware sont des fonctions qui ont accès aux objets requête, réponse et au middleware suivant dans le cycle requête-réponse.

// Middleware pour parser le JSON
app.use(express.json());

// Middleware pour les fichiers statiques
app.use(express.static('public'));

// Middleware personnalisé
const logger = (req, res, next) => {
  console.log(`${new Date().toISOString()} - ${req.method} ${req.url}`);
  next();
};
app.use(logger);

// Middleware pour la gestion d'erreurs
app.use((err, req, res, next) => {
  console.error(err.stack);
  res.status(500).send('Une erreur est survenue!');
});

API REST avec Express

Création d'une API REST

let users = [
  { id: 1, name: 'Alice', email: 'alice@example.com' },
  { id: 2, name: 'Bob', email: 'bob@example.com' }
];

// GET /api/users
app.get('/api/users', (req, res) => {
  res.json(users);
});

// GET /api/users/:id
app.get('/api/users/:id', (req, res) => {
  const user = users.find(u => u.id === parseInt(req.params.id));
  if (!user) return res.status(404).json({ message: 'Utilisateur non trouvé' });
  res.json(user);
});

// POST /api/users
app.post('/api/users', (req, res) => {
  const newUser = {
    id: users.length + 1,
    name: req.body.name,
    email: req.body.email
  };
  users.push(newUser);
  res.status(201).json(newUser);
});

// PUT /api/users/:id
app.put('/api/users/:id', (req, res) => {
  const user = users.find(u => u.id === parseInt(req.params.id));
  if (!user) return res.status(404).json({ message: 'Utilisateur non trouvé' });

  user.name = req.body.name || user.name;
  user.email = req.body.email || user.email;
  res.json(user);
});

// DELETE /api/users/:id
app.delete('/api/users/:id', (req, res) => {
  users = users.filter(u => u.id !== parseInt(req.params.id));
  res.json({ message: 'Utilisateur supprimé' });
});

Authentification JWT

const jwt = require('jsonwebtoken');
const bcrypt = require('bcrypt');

// Route de connexion
app.post('/api/login', async (req, res) => {
  const { email, password } = req.body;
  // Vérifier les identifiants...
  const user = { id: 1, email: 'test@example.com' };
  const token = jwt.sign({ userId: user.id }, 'SECRET_KEY', { expiresIn: '1h' });
  res.json({ token });
});

// Middleware d'authentification
const auth = (req, res, next) => {
  const token = req.header('Authorization');
  if (!token) return res.status(401).send('Accès refusé');

  try {
    const decoded = jwt.verify(token.replace('Bearer ', ''), 'SECRET_KEY');
    req.user = decoded;
    next();
  } catch (err) {
    res.status(400).send('Token invalide');
  }
};

// Route protégée
app.get('/api/protected', auth, (req, res) => {
  res.send('Contenu protégé accessible');
});
